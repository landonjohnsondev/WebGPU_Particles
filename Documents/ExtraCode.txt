for(var i = 0u; i < 3; i++)
							{
								if(uni.isSmoke == 1u || uni.isFire == 1u)
								{
									break;
								}

								var distToCenter = distance(vOutput.worldSpace, uni.explodePos[i]);
								if(distToCenter < 10)
								{												
									//Final Formula: point + normalizedDir * distToEdge
									//new point should be 8 units away!
									
									var normalizedDir = normalize(vOutput.worldSpace - uni.explodePos[i]);
									var distToEdge: f32 = (10 - distToCenter);

									let offset: vec3f = normalizedDir * distToEdge;								
									vOutput.worldSpace += offset;
																		
									vOutput.position = PerspectiveProjection(MoveCamera(vec4f(vOutput.worldSpace, 1)));
								}
							}
